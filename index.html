import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# Definir los vectores de longitud (metros) y tiempo (segundos)
longitud = np.array([0, 10, 20, 30, 40, 50])  # en metros
tiempo = np.array([0, 2, 4, 6, 8, 10])        # en segundos

# Calcular la velocidad (metros/segundo)
velocidad = np.diff(longitud) / np.diff(tiempo)

# Graficar la función velocidad en función del tiempo
plt.figure(figsize=(10, 6))
plt.plot(tiempo[1:], velocidad, marker='o', linestyle='-', color='b')
plt.title('Velocidad en función del tiempo')
plt.xlabel('Tiempo (s)')
plt.ylabel('Velocidad (m/s)')
plt.grid(True)
plt.show()

# Graficar en 3D: Longitud, Tiempo, y Velocidad
fig = plt.figure(figsize=(10, 6))
ax = fig.add_subplot(111, projection='3d')
ax.plot3D(tiempo[1:], longitud[1:], velocidad, marker='o', linestyle='-', color='r')
ax.set_title('Gráfico 3D de Velocidad, Longitud y Tiempo')
ax.set_xlabel('Tiempo (s)')
ax.set_ylabel('Longitud (m)')
ax.set_zlabel('Velocidad (m/s)')
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# Definir el vector de radios en cm (mínimo 5 datos)
radios = np.array([1, 2, 3, 4, 5, 6])  # Ejemplo: Radios en centímetros

# Definir el valor de Pi
pi = 3.1416

# Calcular el área para diferentes radios
areas = pi * (radios ** 2)

# Graficar la función Área en función de los radios en 2D
plt.figure(figsize=(10, 6))
plt.plot(radios, areas, marker='o', linestyle='-', color='g')
plt.title('Área en función del Radio')
plt.xlabel('Radio (cm)')
plt.ylabel('Área (cm²)')
plt.grid(True)
plt.show()

# Graficar en 3D: Radio y Área
fig = plt.figure(figsize=(10, 6))
ax = fig.add_subplot(111, projection='3d')

# Ajustar las dimensiones de los datos para la gráfica 3D
R, A = np.meshgrid(radios, areas)

# Graficar la relación en 3D
